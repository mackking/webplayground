Introduction

Mixins

Mixins allow for

Example


@mixin roundbox($width, $bcolor, $bwidth: 5px, $margin: 0.5em, $pad: 1em, $radius: 10px) {
	margin: $margin;
	padding: $pad;
	width: $width;
	border: $bwidth solid $bcolor;
	border-radius: $radius;
	background-color: lighten($bcolor, 25%);
}

.orangebox {
	@include roundbox(20em, orange, $radius: 25px);
}
.redbox {
	@include roundbox(15%, red); 
}
.greebox {
	@include roundbox(300px, green, 10px, $radius: 50px); 
}
